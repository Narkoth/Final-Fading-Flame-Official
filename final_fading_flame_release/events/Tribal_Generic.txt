#Deciding wether or not to be expansionist or focus on inward development
country_event = {
	id = tribal_generic.3
	title = tribal_generic.3.t
	desc = tribal_generic.3.d
	
	picture = GFX_report_event_military_planning

	is_triggered_only = yes

	option = {
		name = tribal_generic.3.a
		ROOT = {
            add_ideas = tribal_expansionist
        }
        random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
                    is_capital = yes
					free_building_slots = {
						building = arms_factory
						size > 0
						include_locked = yes
					}
				}
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = arms_factory
					level = 1
					instant_build = yes
				}
			}
	}
    option = {
		name = tribal_generic.3.b
        ROOT = {
            add_ideas = tribal_improve_nation
        }
		random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
                    is_capital = yes
					free_building_slots = {
						building = industrial_complex
						size > 0
						include_locked = yes
					}
				}
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = industrial_complex
					level = 1
					instant_build = yes
				}
			}
	}	
}
