add_namespace = power

#Great Power Fact > 250 Factories Army Manpower > 3.500.000
country_event = {
	id = power.1
	title = power.1.t
	desc = power.1.d

	trigger = {
		always = no
		num_of_factories > 250
		NOT = {
			has_idea = great_power
		}
	}

	option = {
		name = power.1.a
		add_ideas = great_power
		remove_ideas = secondary_power
		remove_ideas = regional_power
		remove_ideas = minor_power
		remove_ideas = irrelevant_power
	}
}
