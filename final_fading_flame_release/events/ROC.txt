add_namespace = ROC_general_events



# Aid Sent
country_event = {
	id = ROC_general_events.1
	title = "Ronard Declared War on the Iron State" #Can also be set in localization
	desc = ROC_general_events.1.d
	picture = GFX_report_event_chinese_soldiers_fighting
	
	is_triggered_only = yes
	
	option = { 
		name = "They have gone to far, offer an alliance immediately"
		ai_chance = { 
			factor = 100
		}
		DOC = {
			if = {
				limit = {is_in_faction = yes}
				add_to_faction = CIS
			}
			if = {
				limit = {is_in_faction = no}
				create_faction = "The Kingdom"
				add_to_faction = CIS
			}
		}
		CIS = {
			add_ai_strategy = {
				type = alliance
				id = "DOC"
				value = 200
			}
		}
	}
}
